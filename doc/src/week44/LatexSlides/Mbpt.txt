\section{Lecture 1: Hamiltonian, Reference, and Normal Ordering}
% ============================================================

\begin{frame}{Many-Body Hamiltonian and Decomposition}
We consider a two-body Hamiltonian in second quantization
\[
  \Hh = \sum_{pq} t_{pq} a_p\ad a_q + \frac{1}{4}\sum_{pqrs} \vvbar_{pqrs}\, a_p\ad a_q\ad a_s a_r,\qquad \vvbar_{pqrs} = v_{pqrs}-v_{pqsr}.
\]
Choose a single-determinant reference $\ket{\Phi_0}$ (e.g., Hartree--Fock) and split
\[
  \Hh = \Hzero + \V,\qquad \Hzero \ket{\Phi_0}=\Ezero\ket{\Phi_0},\quad \V=\Hh-\Hzero.
\]
\end{frame}

\begin{frame}{Normal Ordering w.r.t. $\ket{\Phi_0}$}
Using Wick's theorem with respect to $\ket{\Phi_0}$
\[
  \Hh = \eref + \FN + \VN,
\]
where $\eref = \braket{\Phi_0|\Hh|\Phi_0}$, $\FN$ is the normal-ordered one-body part (Fock), and $\VN$ is the normal-ordered two-body residual interaction. Throughout, we adopt hole indices $i,j,\ldots$ (occupied) and particle indices $a,b,\ldots$ (unoccupied).
\end{frame}

\begin{frame}{Projection Operators and Model Space}
Define projectors
\[
  \Pproj = \ket{\Phi_0}\!\bra{\Phi_0},\qquad \Qproj = 1-\Pproj.
\]
The exact ground state satisfies $\Hh \ket{\Psi}=E\ket{\Psi}$. We write the wave operator $\Ohat$ such that
\[
  \ket{\Psi}=\Ohat\ket{\Phi_0},\quad \Pproj \Ohat \Pproj = \Pproj,\quad \Qproj \Ohat \Pproj = \chi.
\]
Rayleigh--Schr\"odinger MBPT is obtained by expanding $\chi$ (or $\Ohat$) in powers of $\V$ with resolvent $\Gzero=\Qproj (\Ezero-\Hzero)^{-1}$.
\end{frame}

  \item Downward line: hole $i,j,\ldots$
  \item Upward line: particle $a,b,\ldots$
  \item Solid dot: two-body vertex $\vvbar_{pqrs}$
  \item Dashed connector: contracted interaction (bookkeeping)
  \item Energy denominators from sums of single-particle energies $\eps_p$ of intermediate states
\end{itemize}

\section{Lecture 2: RS-MBPT Expansions and Energy Corrections}
% ============================================================

\begin{frame}{Wavefunction and Energy Expansions}
Introduce a bookkeeping parameter $\lambda$ and expand
\[
  \ket{\Psi}=\sum_{n\ge0}\lambda^n \ket{\Psi^{(n)}},\qquad E=\sum_{n\ge0}\lambda^n E^{(n)},\qquad (\Hzero+\lambda\V)\ket{\Psi}=E\ket{\Psi}.
\]
At order $n\ge1$,
\[
  (\Hzero-\Ezero)\ket{\Psi^{(n)}} = -\Qproj\Big(\V\ket{\Psi^{(n-1)}}-\sum_{k=1}^{n-1}E^{(k)}\ket{\Psi^{(n-1-k)}}\Big).
\]
With intermediate normalization $\braket{\Phi_0|\Psi}=1$, we get $E^{(n)}=\braket{\Phi_0|\V|\Psi^{(n-1)}}$.
\end{frame}

\begin{frame}{First and Second Order}
Zeroth order: $\ket{\Psi^{(0)}}=\ket{\Phi_0}$, $\Ezero=\braket{\Phi_0|\Hzero|\Phi_0}$.\\
First order energy and amplitudes:
\[
  \Eone=\braket{\Phi_0|\V|\Phi_0},\qquad \ket{\Psi^{(1)}}=\Gzero \V \ket{\Phi_0}.
\]
For a two-body $\V$, singles vanish for a HF reference (Brillouin condition), so only doubles $\ket{\Phi_{ij}^{ab}}$ contribute beyond first order.\\
Second order energy (RS-MBPT(2)):
\[
  \Etwo=\frac{1}{4}\sum_{ijab} \frac{|\vvbar_{ijab}|^2}{\eps_i+\eps_j-\eps_a-\eps_b}.
\]
\end{frame}

\begin{frame}{Third Order (Representative Forms)}
Third order collects linked topologies with two intermediate propagations. A compact representative contribution is
\[
  \Ethree=\frac{1}{8}\sum_{ijkabc} \frac{\vvbar_{ijab}\,\vvbar_{abkc}\,\vvbar_{kcij}}{(\eps_i+\eps_j-\eps_a-\eps_b)(\eps_i+\eps_j-\eps_c-\eps_k)} + \text{permutations},
\]
with additional ring and ladder topologies. All terms are \emph{linked} when expressed in normal-ordered form.
\end{frame}

\begin{frame}{Feynman--Goldstone Rules (Time-Independent)}
\begin{enumerate}
  \item Draw all distinct (topologically unique) diagrams producing a given order in $\V$.
  \item Assign hole lines downward and particle lines upward; vertices carry $\vvbar_{pqrs}$ and conserve lines.
  \item Sum over repeated particle and hole indices; include symmetry factors from identical interchanges.
  \item For each intermediate state, include an energy denominator equal to the difference of hole and particle sums.
  \item Each closed fermion loop contributes a factor $(-1)$.
  \item Overall sign from the parity of line crossings (or from systematic normal-ordering conventions).
\end{enumerate}
\end{frame}

\section{Lecture 4: Systematic Derivation of MBPT(2) and MBPT(3)}
% ============================================================

\begin{frame}{From Resolvents to Diagrammatic Denominators}
Using $\ket{\Psi^{(1)}}=\Gzero\V\ket{\Phi_0}$ and $\ket{\Psi^{(2)}}=\Gzero\big(\V\ket{\Psi^{(1)}}-\Eone\ket{\Psi^{(1)}}\big)$, the second- and third-order energies follow as
\[
 \Etwo=\braket{\Phi_0|\V\Gzero\V|\Phi_0},\qquad
 \Ethree=\braket{\Phi_0|\V\Gzero\big(\V-\Eone\big)\Gzero\V|\Phi_0}.
\]
Insertion of complete sets of $k$-tuple excitations yields Goldstone denominators as differences of single-particle sums.
\end{frame}

\begin{frame}{Explicit MBPT(2) Expression (HF Reference)}
Only doubles contribute:
\[
 \Etwo=\frac{1}{4}\sum_{ijab} \frac{|\vvbar_{ijab}|^2}{\eps_i+\eps_j-\eps_a-\eps_b}.
\]
\begin{itemize}
  \item Factor $\tfrac14$ from antisymmetry and index permutations $(ij)$ and $(ab)$.
  \item Denominator $\eps_i+\eps_j-\eps_a-\eps_b$ from intermediate two-particle--two-hole state.
\end{itemize}
\end{frame}

\begin{frame}{Representative MBPT(3) Forms}
\small
Ladder-type (pp) and ring-type (ph) contributions can be written schematically as
\begin{align*}
E^{(3)}_{\text{lad}} &= \frac{1}{8}\sum_{ijkabc}
\frac{\vvbar_{ijab}\,\vvbar_{abkc}\,\vvbar_{kcij}}{(\eps_i+\eps_j-\eps_a-\eps_b)(\eps_i+\eps_j-\eps_c-\eps_k)} \, ,\\
E^{(3)}_{\text{ring}} &= -\frac{1}{8}\sum_{ijkab}
\frac{\vvbar_{iajb}\,\vvbar_{jbka}\,\vvbar_{kbii}}{(\eps_i-\eps_a+\eps_j-\eps_b)(\eps_i-\eps_a+\eps_k-\eps_b)} \, ,
\end{align*}
plus exchange permutations ensuring full antisymmetry and uniqueness of topologies.
\end{frame}

\section{Lecture 5: Linked-Cluster Theorem (Goldstone) -- Full Derivation}
% ============================================================

\begin{frame}{Statement of the Linked-Cluster Theorem}
For a normal-ordered Hamiltonian with respect to a reference determinant $\ket{\Phi_0}$, the Rayleigh--Schr\"odinger MBPT expansion of the ground-state energy contains only \emph{linked} (connected) diagrams. Unlinked (disconnected) diagrams cancel identically order by order when intermediate normalization is used. Consequently, the energy is size-extensive.
\end{frame}

\begin{frame}{Strategy of the Proof}
\begin{enumerate}
  \item Express $E=\braket{\Phi_0|\Hh\Ohat|\Phi_0}$ with $\Ohat=1+\chi$ and intermediate normalization $\braket{\Phi_0|\Psi}=1$.
  \item Normal order $\Hh=\eref+\FN+\VN$; only $\eref$ and linked contractions survive in the vacuum expectation values.
  \item Use Wick's theorem to organize contractions into \emph{connected} and \emph{disconnected} parts; show that disconnected parts factorize.
  \item Show cancellation of unlinked contributions between numerator and normalization (or between terms generated by $\Eone,\Etwo,\ldots$ in RS recursion).
  \item Conclude that $E-\eref$ is the sum of \emph{connected} (linked) vacuum diagrams only, implying size-extensivity.
\end{enumerate}
\end{frame}

\begin{frame}{Wave Operator and Normalization}
Write $\Ohat=1+\sum_{n\ge1} \Ohat^{(n)}$, where $\Ohat^{(n)}=\big(\Gzero\V\big)^n$ symbolically in RS-MBPT. The exact state is $\ket{\Psi}=\Ohat\ket{\Phi_0}$ with $\braket{\Phi_0|\Psi}=1$. The energy is
\[
  E = \frac{\braket{\Phi_0|\Hh\Ohat|\Phi_0}}{\braket{\Phi_0|\Ohat|\Phi_0}}.
\]
Intermediate normalization enforces $\braket{\Phi_0|\Ohat|\Phi_0}=1$, but the RS recursion effectively generates subtraction terms $E^{(k)}\ket{\Psi^{(n-k)}}$ that mimic the denominator and cancel unlinked pieces.
\end{frame}

\begin{frame}{Cumulant (Connected) Expansion via Wick's Theorem}
Consider the generating series built from normal-ordered interactions acting on $\ket{\Phi_0}$:
\[
  \mathcal{Z} = \braket{\Phi_0| \Nord\exp\big( \lambda\,\VN\big) |\Phi_0 }.
\]
By Wick's theorem, vacuum expectations of exponentials factorize into exponentials of \emph{connected} contractions (cumulants):
\[
  \mathcal{Z} = \exp\big( \mathcal{W}_c(\lambda) \big),\qquad \mathcal{W}_c(\lambda)=\sum_{n\ge1}\frac{\lambda^n}{n!}\,\langle (\VN)^n \rangle_c.
\]
Here $\langle\cdots\rangle_c$ denotes the sum of \emph{connected} vacuum diagrams only. Thus the logarithm selects connected pieces exclusively.
\end{frame}

\begin{frame}{From Generating Functional to Energy}
In time-independent RS-MBPT, the energy corrections can be organized such that
\[
  E-\eref = \sum_{n\ge1} E^{(n)} = \text{(sum of connected vacuum diagrams at order $n$)}.
\]
Sketch: Expand $\Ohat$ order by order and evaluate $\braket{\Phi_0|\VN\Ohat|\Phi_0}$. Disconnected pieces factor into products of lower-order vacuum bubbles. These are canceled by the RS subtraction terms involving lower-order $E^{(k)}$ (which themselves equal the connected bubbles of order $k$). Therefore only linked diagrams remain.
\end{frame}

\begin{frame}{Explicit Cancellation at Low Orders}
\textbf{Second order:} There is no unlinked contribution; $\Etwo$ is a single linked bubble.\\
\textbf{Third order:} Disconnected possibilities are products like $(\text{first-order bubble})\times(\text{second-order bubble})$. RS recursion produces $-\Eone\,\braket{\Phi_0|\Gzero\V\Gzero\V|\Phi_0}$ terms and $-\Etwo\,\braket{\Phi_0|\Gzero\V|\Phi_0}$ that cancel these products. What remains are linked topologies (ring, ladder, etc.).
\end{frame}

\begin{frame}{Size-Extensivity}
For two noninteracting subsystems $A$ and $B$ (block-diagonal $\Hh$), the connected diagrams are confined within each block. Hence $E(A\oplus B) = E(A)+E(B)$, establishing size-extensivity. Disconnected vacuum products would violate this property; their cancellation is thus essential.
\end{frame}

\section{Lecture 6: Perspectives and Advanced Connections}
% ============================================================

\begin{frame}{Connections to Coupled Cluster}
Coupled cluster writes $\ket{\Psi}=e^{\hat T}\ket{\Phi_0}$ with $\hat T=T_1+T_2+\cdots$. The linked-cluster property is built-in: $E=\braket{\Phi_0|e^{-\hat T}\Hh e^{\hat T}|\Phi_0}$ contains only connected (linked) diagrams. MBPT is recovered by truncating $\hat T$ order by order.
\end{frame}

\begin{frame}{Connections to Green's Functions}
Single-particle Green's functions and the self-energy $\Sigma(\omega)$ resum infinite classes of linked diagrams (ladders, rings). The MBPT energy can be related to linked vacuum diagrams which are the zero-source limit of generating functionals; the Luttinger--Ward functional viewpoint organizes only connected contributions.
\end{frame}

\begin{frame}{Summary of the Six Lectures}
\begin{itemize}
  \item Established RS-MBPT with a normal-ordered Hamiltonian and a single-determinant reference.
  \item Derived explicit MBPT(2) and representative MBPT(3) expressions with Goldstone denominators.
  \item Built and used Feynman--Goldstone rules; drew key second- and third-order topologies.
  \item Proved the linked-cluster theorem via cumulant expansion and explicit RS cancellations.
  \item Highlighted size-extensivity and connections to advanced many-body formalisms.
\end{itemize}
\end{frame}

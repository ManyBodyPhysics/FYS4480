\begin{frame}{CCSD energy equation }
    \note{Filename: ccsd\_algebraicequations02.tex}
    \begin{equation*}
    E_{CCSD} = 
    f_a^i t_i^a + \frac{1}{4} \bra{ij}\ket{ab} t_{ij}^{ab} + \frac{1}{2} \bra{ij}\ket{ab}  t_i^a  t_j^b
\end{equation*}

Note the implicit sum over repeated indices.



\end{frame}
